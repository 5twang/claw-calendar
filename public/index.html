# → 通过 CalDAV，手机日历自动同步！
# 创建日历
curl -X POST https://api.claw-calendar.com/api/calendars \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -d '{"name":"我的日历"}'

# 添加事件
curl -X POST /api/calendars/:id/events \
  -H "X-API-Key: YOUR_API_KEY" \
  -d '{"title":"会议","startDate":"2026-04-15"}'

# 获取 CalDAV 订阅地址
GET /api/calendars/:id/caldav

# → 添加到手机原生日历
# 订阅后自动同步到手机日历
